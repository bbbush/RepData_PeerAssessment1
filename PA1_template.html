Imputing missing values

nas<-is.na(df$steps)
sum(nas)
## [1] 2304
# use mean for the interval
stepsImputed<-df$steps
stepsImputed[is.na(stepsImputed)]<-0
stepsImputed<-stepsImputed+dm[interval==df$interval]$V1*nas
